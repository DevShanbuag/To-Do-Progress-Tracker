// Example frontend API call
const response = await fetch('http://localhost:5000/api/auth/register', {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json',
    },
    body: JSON.stringify({
        username,
        email,
        password
    })
});

if (!response.ok) {
    const error = await response.json();
    throw new Error(error.message);
}
